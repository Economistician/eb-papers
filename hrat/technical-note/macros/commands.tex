% ----------------------------------------------------------
% METRIC MACROS (FRF / Electric Barometer)
% ----------------------------------------------------------

% Metric name (upright math font)
\newcommand{\HRAT}{\ensuremath{\mathrm{HR@\tau}}}

% Tolerance parameter
\newcommand{\tauTol}{\ensuremath{\tau}}

% ----------------------------------------------------------
% FORECAST / DEMAND NOTATION
% ----------------------------------------------------------

% Forecast value \hat{y}
\newcommand{\yhat}{\hat{y}}

% Absolute error |e|
\newcommand{\abserr}{\left| e \right|}

% Interval-level error e_t
\newcommand{\errt}{e_t}

% ----------------------------------------------------------
% HIT / MISS INDICATORS
% ----------------------------------------------------------

% Hit indicator: 1 if |e_t| <= tau
\newcommand{\hit}{\ensuremath{\mathbb{I}\{ |\errt| \le \tauTol \}}}

% Miss indicator (optional)
\newcommand{\miss}{\ensuremath{\mathbb{I}\{ |\errt| > \tauTol \}}}

% ----------------------------------------------------------
% SET / COUNTING OPERATORS
% ----------------------------------------------------------

% Hit set: { t in T : |e_t| <= tau }
\newcommand{\hitset}{\ensuremath{\{ t \in T : |\errt| \le \tauTol \}}}

% Number of evaluation intervals
\newcommand{\Tsize}{\ensuremath{|T|}}

% Cardinality operator
\newcommand{\card}[1]{\left| #1 \right|}
